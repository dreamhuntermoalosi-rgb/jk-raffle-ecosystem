  1762→/**
  1763→ * Returns 4 featured campaigns for homepage display
  1764→ */
  1765→export function getFeaturedCampaigns(): Campaign[] {
  1766→  const openFeatured = mockCampaigns
  1767→    .filter(c => c.status === 'open' && c.product.featured)
  1768→    .sort((a, b) => b.progress - a.progress)
  1769→    .slice(0, 4);
  1770→
  1771→  // If we don't have 4 featured, fill with top open campaigns
  1772→  if (openFeatured.length < 4) {
  1773→    const remaining = mockCampaigns
  1774→      .filter(c => c.status === 'open' && !openFeatured.includes(c))
  1775→      .sort((a, b) => b.progress - a.progress);
  1776→    openFeatured.push(...remaining.slice(0, 4 - openFeatured.length));
  1777→  }
  1778→
  1779→  return openFeatured.slice(0, 4);
  1780→}

  1795→/**
  1796→ * Returns computed stats for a specific branch
  1797→ */
  1798→export function getBranchStats(branchId: string) {
  1799→  const branch = mockBranches.find(b => b.id === branchId);
  1800→  if (!branch) return null;
  1801→
  1802→  const branchCampaigns = mockCampaigns.filter(c => c.branchId === branchId);
  1803→  const activeCampaigns = branchCampaigns.filter(c => c.status === 'open');
  1804→  const completedCampaigns = branchCampaigns.filter(c => c.status === 'completed');
  1805→
  1806→  const totalTicketsSold = branchCampaigns.reduce((sum, c) => sum + c.soldTickets, 0);
  1807→  const totalRevenue = branchCampaigns.reduce((sum, c) => sum + (c.soldTickets * c.ticketPrice), 0);
  1808→
  1809→  return {
  1810→    branch,
  1811→    activeCampaignCount: activeCampaigns.length,
  1812→    completedCampaignCount: completedCampaigns.length,
  1813→    totalTicketsSold,
  1814→    totalRevenue,
  1815→    averageProgress: activeCampaigns.length > 0
  1816→      ? Math.round(activeCampaigns.reduce((sum, c) => sum + (c.progress || 0), 0) / activeCampaigns.length)
  1817→      : 0,
  1818→    upcomingDraws: activeCampaigns
  1819→      .sort((a, b) => new Date(a.drawDate).getTime() - new Date(b.drawDate).getTime())
  1820→      .map(c => ({
  1821→        campaignTitle: c.title,
  1822→        drawDate: c.drawDate,
  1823→        progress: c.progress || 0,
  1824→        ticketsRemaining: c.maxTickets - c.soldTickets,
  1825→      })),
  1826→  };
  1827→}
